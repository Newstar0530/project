# step1
import pandas as pd
import matplotlib.pyplot as plt

# step2
df = pd.read_csv('/Users/newstar0530/python/ML learning/pandas/titanic.csv')

# step3
# Display the first few rows of the dataset to understand its structure
df.head()

# Print information about the columns and their data types
df.info()

# Show summary statistics for numerical columns
df.describe()

# step4
# compute median of column Age
median = df['Age'].median()
# let median fill in missing values
df['Age'].fillna(median, inplace=True)

# step5
# Calculate and display the count of passengers by gender
df['Sex'].value_counts()

# Compute the average age of passengers
df['Age'].mean()

# Determine the survival rate by passenger class
survival_rate_by_class = df.groupby('Pclass')['Survived'].mean()
print(survival_rate_by_class)

#step6
# Create a bar chart
plt.bar(survival_rate_by_class.index, survival_rate_by_class.values)

# Add labels and title
plt.xlabel('Passenger Class')
plt.ylabel('Survival Rate')
plt.title('Survival Rate by Passenger Class')

# Show the plot
plt.show()
